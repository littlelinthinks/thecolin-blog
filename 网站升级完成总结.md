## ✅ 模块6：用户体验优化

### 6.1 实时搜索功能
```javascript
function searchProducts() {
    const searchText = document.getElementById('searchInput').value.toLowerCase();
    const products = document.querySelectorAll('.product');

    products.forEach(product => {
        // 搜索标题、描述、特性
        const title = product.querySelector('.product-title')?.textContent.toLowerCase() || '';
        const desc = product.querySelector('.product-desc')?.textContent.toLowerCase() || '';
        const features = Array.from(product.querySelectorAll('.feature-text'))
            .map(f => f.textContent.toLowerCase())
            .join(' ');

        if (title.includes(searchText) || desc.includes(searchText) || features.includes(searchText)) {
            product.style.display = 'block';
        } else {
            product.style.display = searchText === '' ? 'block' : 'none';
        }
    });
}
```
